3. 다음과 같은 동작을 하는 JavaScript 코드를 작성하시오.

사용자에게 "이 페이지를 보시겠습니까?" 라는 질문을 띄웁니다.
사용자가 확인을 누르면, "환영합니다!" 라는 문구를 웹 페이지에 출력합니다.
사용자가 취소를 누르면, "다음에 다시 방문해주세요." 라는 문구를 웹 페이지에 출력합니다.

let b = confirm("이 페이지를 보시겠습니까?");
if(b){
	document.write("<p>환영합니다!</p>");
}
else{
	document.write("<p>다음에 다시 방문해주세요.</p>");
}

6. 아래 코드를 실행했을 때 출력 결과와 오류 발생 여부를 작성하시오.

function testScope() {
  var a = 10;
  let b = 2;
  var c = 30;

  if (true) {
    var a = 20;
    let b = 3;
    var c = 40;
    let d = 4;
    console.log(a); // (1)
    console.log(b); // (2)
    console.log(c); // (3)
    console.log(d); // (4)
  }

  console.log(a); // (5)
  console.log(b); // (6)
  console.log(c); // (7)
  console.log(d); // (8)
}

testScope();
