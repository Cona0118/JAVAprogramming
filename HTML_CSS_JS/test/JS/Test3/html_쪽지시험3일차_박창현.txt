1. HTML과 JavaScript를 사용하여 다음 기능을 구현하시오.

화면에 버튼 하나를 만들고, 그 버튼에 계산하기라는 문구를 표시하시오.
사용자가 버튼을 클릭하면 prompt() 창을 두 번 띄워 두 개의 숫자를 입력받으시오.
입력받은 두 숫자를 더한 값을 alert()을 사용하여 알림창으로 출력하시오.
계산하는 코드는 반드시 함수로 작성하시오.
(이벤트에 관한 내용는 버튼태그의 속성을 이용하여 작성)

<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="UTF-8" />
  <title>Q1</title>
</head>
<body>
  <button>계산하기</button>

  <script>
    let cal = document.querySelector("button");

    cal.addEventListener("click", function(){
      let num1 = parseInt(prompt("숫자를 입력하세요:"));
      let num2 = parseInt(prompt("숫자를 입력하세요:"));
      alert(`${num1} + ${num2} = ${sum(num1,num2)}`);
    });

    function sum(n1,n2){
      return n1+n2;
    }
  </script>
</body>
</html>


2. HTML과 JavaScript를 사용하여 다음 조건을 만족하는 프로그램을 작성하시오.

페이지가 열리면 사용자에게 "이름을 입력하세요:" 라고 묻고, 이름을 입력받으시오.
이름이 입력되었을 경우, HTML 페이지에 "안녕하세요? ○○님"이라는 문장을 출력하시오.
사용자가 입력을 하지 않았거나, 취소 버튼을 누른 경우, "이름을 입력하지 않으셨습니다." 라는 문장을 출력하시오. (모든 동작은 즉시 실행 함수 안에서 수행)

<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="UTF-8" />
  <title>Q2</title>
</head>
<body>

  <script>
    document.addEventListener("DOMContentLoaded", function(){
      let name = prompt("이름을 입력하세요: ");

      if (name !== null && name !== ""){
        document.write(`안녕하세요? ${name}님`);
      } else {
        document.write("이름을 입력하지 않으셨습니다.");
      }
    });
  </script>
</body>
</html>


3. HTML과 JavaScript를 사용하여 다음 조건을 만족하는 프로그램을 작성하시오.

화면에 "눌러보세요"라는 문구가 적힌 버튼을 만들고,
사용자가 버튼을 클릭하면 "버튼이 클릭되었습니다."라는 문장을 HTML 페이지에 출력하시오.
(이벤트에 관한 내용은 이벤트 리스너를 이용하여 작성)

<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="UTF-8" />
  <title>Q3</title>
</head>
<body>
  <button>눌러보세요</button>

  <div id="message"></div>

  <script>
    let button = document.querySelector("button");

    button.addEventListener("click", function(){
      let page = document.querySelector("#message");

      page.textContent = "버튼이 클릭되었습니다.";
    });
  </script>
</body>
</html>



4. openButton.html 파일 안에 다음 조건을 만족하는 기능을 구현하시오.

"페이지 열기" 라는 문구가 적힌 버튼을 만들고,
사용자가 버튼을 클릭하면 popup 폴더 안에 있는 optionPage.html 파일을 브라우저에서 열리도록 구현하시오. (경로는 상대 경로로 작성하며, addEventListener()를 사용, 팝업창은 window함수 사용)

project/
├── popup/
│   └── optionPage.html    ← 팝업창으로 띄울 HTML 파일
└── pages/
    └── sub/
        └── openButton.html   ← 작업하고 있는 HTML 파일

<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="UTF-8" />
  <title>Q4</title>
</head>
<body>
  <button>페이지 열기</button>

  <script>
    let button = document.querySelector("button");
    button.addEventListener("click", function(){
      window.open("../../popup/optionPage.html","popup","width=500, height=400")
    });
  </script>
</body>
</html>

5. HTML과 JavaScript를 이용하여 다음 조건을 만족하는 프로그램을 작성하시오.

<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="UTF-8" />
  <title>querySelector 다양한 선택자</title>
</head>
<body>

  <h1 id="title">안녕하세요!</h1>
  <p class="desc">설명문구입니다.</p>
  <button>클릭하세요</button>

  <script>
    let button = document.querySelector("button");
    button.addEventListener("click",function(){
      document.querySelector("#title").textContent = "환영합니다!";
      document.querySelector(".desc").textContent = "버튼이 클릭되었습니다.";
    });
  </script>

</body>
</html>

버튼을 클릭하면 아래 동작을 수행하시오.

<h1> 텍스트를 "환영합니다!" 로 변경
<p> 텍스트를 "버튼이 클릭되었습니다." 로 변경
